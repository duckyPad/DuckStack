
FUNCTION print_result(value)
    STRINGLN Value is: $value
END_FUNCTION

FUNCTION is_even(n)
    VAR mylocal = n%2
    IF mylocal == 0 THEN
        RETURN 1
    END_IF
    RETURN 0
END_FUNCTION

VAR i = 1
VAR sum = 0

WHILE i <= 6
    IF is_even(i) THEN
        sum = sum + i
    END_IF
    i = i + 1
END_WHILE

print_result(sum)
