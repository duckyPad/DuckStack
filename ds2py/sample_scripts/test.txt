// 1. Declare Global Variables
VAR myVar = 100
VAR counter = 1

// 2. Define Function with argument naming conflict
FUNCTION scopeCheck(myVar)
    // Print Argument (Should be the value passed in: 50)
    STRINGLN [FUNC] Arg myVar: $myVar

    // 3. Declare Local Variable (Shadowing the Argument AND Global)
    // Note: In C-like langs, you usually can't shadow an arg in the same scope, 
    // but in dynamic stack scripting, this often overwrites the local stack slot 
    // or creates a new one. For this test, let's declare a unique local 
    // to test strict local vs global.
    
    VAR localVar = 777 + myVar
    
    // Modify the local variable
    localVar = localVar + 1
    
    // Modify the global variable 'counter' from inside function
    counter = counter + 1

    STRINGLN [FUNC] LocalVar: $localVar
    STRINGLN [FUNC] Global Counter: $counter
    
    RETURN
END_FUNCTION

// --- Main Execution ---

STRINGLN [MAIN] Start Global: $myVar
STRINGLN [MAIN] Start Counter: $counter

// Call function with value 50
scopeCheck(50)

// Check if Global was protected from argument/local changes
STRINGLN [MAIN] End Global: $myVar

// Check if Global Counter was successfully modified by function
STRINGLN [MAIN] End Counter: $counter
