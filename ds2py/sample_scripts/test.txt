REM Compiler Workout Script

VAR global_res = 0

// Function to calculate factorial
FUNCTION factorial(n)
    VAR acc = 1
    VAR i = 1
    
    WHILE i <= n
        acc = acc * i
        i = i + 1
    END_WHILE
    
    RETURN acc
END_FUNCTION

STRINGLN Starting Calculation...

// Test Function Call & Assignment
global_res = factorial(5)

// Test Advanced Printing (Hex mode)
_STR_PRINT_FORMAT = 3
STRINGLN Result (0x78 expected): $global_res

// Test Conditionals & Key Output
IF global_res == 120 THEN
    STRINGLN Test Passed!
    DELAY 500
    WINDOWS r
    DELAY 200
    STRINGLN notepad
END_IF
