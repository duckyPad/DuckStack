VAR global_res = 0
FUNCTION factorial(n)
    VAR acc = 1
    VAR i = 1
    WHILE i <= n
        acc = acc * i
        i = i + 1
    END_WHILE
    RETURN acc
END_FUNCTION
STRINGLN Starting Calculation...
global_res = factorial(5)
_STR_PRINT_FORMAT = 3
STRINGLN Result (0x78 expected): $global_res
IF global_res == 120 THEN
    STRINGLN Test Passed!
    DELAY 500
    KEYDOWN WINDOWS
    KEYDOWN r
    KEYUP r
    KEYUP WINDOWS
    DELAY 200
    STRINGLN notepad
END_IF
