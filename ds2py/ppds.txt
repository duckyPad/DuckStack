VAR g = 10
VAR unused_global = 123
FUNCTION no_args()
    VAR local = 1
    VAR unused_local = 999
    RETURN
    local = 42
END_FUNCTION
FUNCTION add(a, b, unused_arg)
    VAR tmp = a + b
    IF tmp == 0 THEN
        RETURN 0
    END_IF
    RETURN tmp
END_FUNCTION
FUNCTION factorial(n)
    VAR acc = 1
    WHILE n > 1
        acc = acc * n
        n = n - 1
        IF n == 2 THEN
            n = n - 1
            CONTINUE
        END_IF
    END_WHILE
    RETURN acc
END_FUNCTION

VAR result = 0

result = add(1, 2, 3)

result = add(g, 1, unused_global)

result = factorial(5)

factorial(0)
WHILE i < 5  and  result != 0
