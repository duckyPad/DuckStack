FUNCTION blink()
    VAR $count = 0
    WHILE $count < 4
        SWC_FILL 100 100 100
        DELAY 200
        SWC_FILL 0 0 0
        DELAY 200
        $count = $count + 1
    END_WHILE
END_FUNCTION
