_UNSIGNED_MATH = 1

_STR_PRINT_PADDING = 4
_DSVM_DUMMY = POKE8(0xfa02, 0x9)

start_addr = 0xfa00
end_addr = start_addr + 5
curr_addr = start_addr
WHILE curr_addr <= end_addr
    value = PEEK8(curr_addr)
    STRINGLN Addr $curr_addr: $value
    curr_addr = curr_addr + 1
END_WHILE
